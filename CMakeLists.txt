cmake_minimum_required(VERSION 3.20)
cmake_path(GET CMAKE_CURRENT_LIST_DIR FILENAME fileName)
cmake_path(ABSOLUTE_PATH CMAKE_CURRENT_LIST_DIR OUTPUT_VARIABLE includePath)
cmake_path(APPEND CMAKE_CURRENT_LIST_DIR include OUTPUT_VARIABLE includePath)
project(${fileName} LANGUAGES C)
set(SHAREDOK SHARED)
set(LIBRARY_OUTPUT_PATH ${CMAKE_SOURCE_DIR}/bin)
set(EXECUTABLE_OUTPUT_PATH ${CMAKE_SOURCE_DIR}/bin)
add_executable(${PROJECT_NAME} demo.c)
include_directories(./cacheconfig ${includePath})
target_link_libraries(${PROJECT_NAME} cacheconfig bisonLex)
add_subdirectory(bisonLex)
add_subdirectory(cacheconfig)
