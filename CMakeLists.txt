set(CMV 3.8)
cmake_minimum_required(VERSION ${CMV})
project(test LANGUAGES C)

include(./utils.cmake)


list(APPEND DEBUGLIST Debug RelWithDebInfo)
isInList(DEBUGLIST ${CMAKE_BUILD_TYPE} RESULT)
if(${RESULT})
    set(SHARED)
else()
    set(SHARED SHARED)
endif()
set(TARGEOUT ${CMAKE_BINARY_DIR})

set(EXECUTABLE_OUTPUT_PATH ${TARGEOUT}/bin)
add_executable(${PROJECT_NAME} test.c)
add_subdirectory(Pdf)
report(${CMAKE_LIBRARY_OUTPUT_DIRECTORY})
